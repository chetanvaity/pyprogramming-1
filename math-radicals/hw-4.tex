    \item $\sqrt{2} \cdot \sqrt{8}$
        \textcolor{blue}{
        \begin{align*}
            \sqrt{2} \cdot \sqrt{8} &= \sqrt{2} \cdot \sqrt{2 \cdot 4} \\
                                    &= \sqrt{2} \cdot (\sqrt{2} \cdot \sqrt{4}) \\
                                    &= \sqrt{2} \cdot \sqrt{2} \cdot \sqrt{4} \\
                                    &= 2 \cdot 2 \\
                                    &= 4
        \end{align*}
