\section*{\normalsize 2. Multiply}
\begin{enumerate}[label=\alph*.]
    \item $\sqrt{x} (2 + 3\sqrt{x})$
    \textcolor{blue}{
        \begin{align*}
            \sqrt{x} (2 + 3\sqrt{x}) &= (\sqrt{x} \cdot 2) + (\sqrt{x} \cdot (3\sqrt{x})) \\
                                     &= 2\sqrt{x} + (3 \cdot \sqrt{x} \cdot {\sqrt{x}}) \\
                                     &= 2\sqrt{x} + 3x
        \end{align*}
    }
